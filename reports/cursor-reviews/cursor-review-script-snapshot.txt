## -- cursor-review — headless deep code review via cursor-agent (RONDO-351/353).
## -- Runs Cursor's agent in PLAN mode (read-only — explores + reviews, NEVER
## -- edits) on the current repo, saves a timestamped report. Works from any
## -- repo / any Claude Code session (~/bin, on PATH).
##
## -- Usage:
## --   cursor-review                       # broad gap/risk review (spec-aware)
## --   cursor-review --lens security       # one focused deep lens
## --   cursor-review --lens concurrency    # (security|concurrency|errors|spec)
## --   cursor-review "free-text focus"     # custom focus
## --   cursor-review --model gpt-5.2 ...   # diversify the reviewer
##
## -- RONDO-353 upgrades that make it find MORE:
## --   * SPEC-AWARE by default — cross-checks every MUST/SHALL in specs/ vs the
## --     code (this is what caught the prompt_sent secret-leak that 3 cloud AIs
## --     missed). Code-vs-spec drift is where the real bugs hide.
## --   * FOCUSED LENSES — one deep pass per failure family beats one broad sweep.
## --   * ANTI-REPEAT — feeds prior reports so it digs for what's LEFT.

setopt ERR_EXIT NO_UNSET PIPE_FAIL

MODEL="claude-opus-4-8-thinking-high"
LENS=""
while [[ "${1:-}" == --* ]]; do
   case "$1" in
      --model) MODEL="$2"; shift 2 ;;
      --lens)  LENS="$2";  shift 2 ;;
      *) print -- "unknown flag: $1"; exit 2 ;;
   esac
done
FOCUS="${1:-}"

REPO="$(git rev-parse --show-toplevel 2>/dev/null || pwd)"
NAME="$(basename "$REPO")"
STAMP="$(date +%Y%m%d-%H%M%S)"
OUTDIR="$REPO/reports/cursor-reviews"
OUT="$OUTDIR/review-${STAMP}${LENS:+-$LENS}.md"
mkdir -p "$OUTDIR"

## -- Lens definitions: each is a sharp, single-family deep pass.
case "$LENS" in
   security)     LENS_TXT="SECURITY ONLY: secret leaks to any persisted artifact (logs/spool/results/audit), fail-open security controls, injection, unsafe subprocess/env, path traversal, auth bypass. Trace every path a secret could reach disk." ;;
   concurrency)  LENS_TXT="CONCURRENCY ONLY: data races on shared/global mutable state, unlocked reads of locked fields, lost updates, budget/cost overruns under parallel workers, lock ordering, TOCTOU. rondo runs a ThreadPool — assume real parallelism." ;;
   errors)       LENS_TXT="ERROR-HANDLING ONLY: swallowed exceptions, silent fallbacks that hide failures, non-zero exit treated as success, partial results recorded as done, bare/broad excepts, error paths that don't reach the envelope." ;;
   spec)         LENS_TXT="SPEC-DRIFT ONLY: read specs/ and find every MUST/SHALL/'guarantee' claim, then prove or disprove it in code. Report each spec requirement that the code does NOT actually satisfy (file:line of the gap + the spec line)." ;;
   "")           LENS_TXT="BROAD: correctness/races, error-handling holes, security/secret-leak risks, the single biggest architectural weakness, and what's genuinely well-built." ;;
   *) print -- "unknown lens: $LENS (use security|concurrency|errors|spec)"; exit 2 ;;
esac

## -- Anti-repeat: tell Cursor what's already been found so it digs deeper.
PRIOR=""
if ls "$OUTDIR"/*.md >/dev/null 2>&1; then
   PRIOR=" Prior reviews exist in reports/cursor-reviews/ — do NOT repeat already-known findings; surface what is LEFT."
fi

PROMPT="Hostile, deep code review of this repository ('$NAME'). Explore the \
codebase yourself, INCLUDING specs/. SPEC-AWARE: cross-check the code against \
every MUST/SHALL/'guarantee' in specs/ and flag any the code does not actually \
satisfy (cite the spec line AND the code file:line). Lens: $LENS_TXT Be \
specific with file:line, rank by severity, top 10 findings.$PRIOR"
[[ -n "$FOCUS" ]] && PROMPT="$PROMPT Extra focus: $FOCUS"

print -- "  cursor-review: $NAME  (model=$MODEL, plan/read-only, lens=${LENS:-broad})"
print -- "  -> $OUT"

{
   print -- "# Cursor deep review — $NAME — $STAMP"
   print -- ""
   print -- "**Model:** $MODEL | **Lens:** ${LENS:-broad} | **Mode:** plan (read-only)"
   print -- ""
   print -- "---"
   print -- ""
} > "$OUT"

if cursor-agent -p "$PROMPT" --mode plan --trust \
      --output-format text --model "$MODEL" >> "$OUT" 2>&1; then
   print -- "  -PASS- review complete"
else
   rc=$?
   print -- "  -ERROR- cursor-agent exited $rc (see $OUT)"
   exit $rc
fi
